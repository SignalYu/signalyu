属性注入#

package org.signalyu.spring6.ioc_anno.pojo;

public class User {
    private Integer id;
    private String name;
}

package org.signalyu.spring6.ioc_anno.dao;

import org.signalyu.spring6.ioc_anno.pojo.User;

public interface UserDao {
    User getUserById(int id);
}

package org.signalyu.spring6.ioc_anno.dao;

import org.signalyu.spring6.ioc_anno.pojo.User;
import org.springframework.stereotype.Repository;

@Repository
public class UserDaoImpl implements UserDao{
    @Override
    public User getUserById(int id) {
        System.out.println("UserDaoImpl: getUserById(" + id +")");
        return null;
    }
}

package org.signalyu.spring6.ioc_anno.service;

import org.signalyu.spring6.ioc_anno.pojo.User;

public interface UserService {
    User getUserById(int id);
}

package org.signalyu.spring6.ioc_anno.service;

import org.signalyu.spring6.ioc_anno.dao.UserDao;
import org.signalyu.spring6.ioc_anno.pojo.User;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service
public class UserServiceImpl implements UserService {
    @Autowired
    private UserDao userDao;

    @Override
    public User getUserById(int id) {
        System.out.println("UserServiceImpl: getUserById(" + id +")");
        return userDao.getUserById(id);
    }
}

package org.signalyu.spring6.ioc_anno.controller;

import org.signalyu.spring6.ioc_anno.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;

@Controller
public class UserController {
    @Autowired
    private UserService userService;

    public String getUserById(int id) {
        System.out.println("Controller: getUserById(" + id + ")");
        userService.getUserById(id);
        return "user";
    }
}

import org.junit.jupiter.api.Test;
import org.signalyu.spring6.ioc_anno.controller.UserController;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

public class UserTest {
    @Test
    public void testUser() {
        ApplicationContext context = new ClassPathXmlApplicationContext("beans.xml");
        UserController userController = context.getBean(UserController.class);
        userController.getUserById(999);
    }
}

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
    xmlns:context="http://www.springframework.org/schema/context"
    xsi:schemaLocation="http://www.springframework.org/schema/beans
    http://www.springframework.org/schema/beans/spring-beans.xsd
    http://www.springframework.org/schema/context
    http://www.springframework.org/schema/context/spring-context.xsd">

    <context:component-scan base-package="org.signalyu.spring6.ioc_anno">
    <!-- context:include-filter标签：指定在原有扫描规则的基础上追加的规则 -->
    <!-- use-default-filters属性：取值false表示关闭默认扫描规则 -->
    <!-- 此时必须设置use-default-filters="false"，因为默认规则即扫描指定包下所有类 -->
    <!--
        type：设置排除或包含的依据
        type="annotation"，根据注解排除，expression中设置要排除的注解的全类名
        type="assignable"，根据类型排除，expression中设置要排除的类型的全类名
    -->
    <context:include-filter type="annotation" expression="org.springframework.stereotype.Controller"/>
    <!---->
</context:component-scan>
</beans>

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xmlns:context="http://www.springframework.org/schema/context"

       http://www.springframework.org/schema/context
       http://www.springframework.org/schema/context/spring-context.xsd">

    <!---->
        <!-- context:exclude-filter标签：指定排除规则 -->
        <!--
            type：设置排除或包含的依据
            type="annotation"，根据注解排除，expression中设置要排除的注解的全类名
            type="assignable"，根据类型排除，expression中设置要排除的类型的全类名
        -->
    <!---->
    <context:exclude-filter type="assignable" expression="org.signalyu.spring6.ioc_anno.controller.UserController"/>
</context:component-scan>
</beans>
指定扫描组件只指定 UserController，由于测试中还用到了 UserServiceImpl 和 UserDaoImpl 组件，因此测试报错。
  • @Component 注解用于描述容器中的一个组件（Bean），可以作用在应用的任何层次，例如 Service 层、Dao 层等。
  • @Repository 注解用于将数据访问层（Dao 层）的类标识为 Spring 中的 Bean，功能与 @Component 相同。
  • @Service 注解用于将数据访问层（Serivce 层）的类标识为 Spring 中的 Bean，功能与 @Component 相同。
  • @Controller 注解用于将数据访问层（Controller 层）的类标识为 Spring 中的 Bean，功能与 @Component 相同。
